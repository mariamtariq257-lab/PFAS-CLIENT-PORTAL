# ===================================================== # PFAS Portal Data Fetcher # Fetches: OneDrive URLs from ClickUp + Teams channel deep-links # Output: ONE JSON file you paste back to Claude # ===================================================== # === SETUP === $ClickUpToken = "pk_107584191_C8D2GPU5C2QPLOQB8U15KBQUDOYVU60O" # Your ClickUp API token $ProjectDirectoryListId = "901817826150" # PFAS Project Directory list $OutputFile = "$HOME/Desktop/pfas_portal_data.json" $Result = @{ onedrive_links = @{} teams_channels = @{} fetched_at = (Get-Date -Format "yyyy-MM-dd HH:mm") } Write-Host "============================================" -ForegroundColor Cyan Write-Host " PFAS Portal Data Fetcher" -ForegroundColor Cyan Write-Host "============================================" -ForegroundColor Cyan Write-Host "" # === PART 1: ClickUp OneDrive URLs === Write-Host "[1/2] Fetching OneDrive URLs from ClickUp..." -ForegroundColor Yellow try { $headers = @{ Authorization = $ClickUpToken } $listUrl = "https://api.clickup.com/api/v2/list/$ProjectDirectoryListId/task?include_closed=true&subtasks=false" $listResp = Invoke-RestMethod -Uri $listUrl -Headers $headers -Method Get Write-Host " Found $($listResp.tasks.Count) projects in Project Directory" -ForegroundColor Green foreach ($task in $listResp.tasks) { # Get detailed task with custom fields $taskUrl = "https://api.clickup.com/api/v2/task/$($task.id)?include_subtasks=false" $taskDetail = Invoke-RestMethod -Uri $taskUrl -Headers $headers -Method Get $oneDrive = ($taskDetail.custom_fields | Where-Object { $_.name -eq "One Drive" }).value $clickupProj = ($taskDetail.custom_fields | Where-Object { $_.name -eq "Clickup Project" }).value $clickupChan = ($taskDetail.custom_fields | Where-Object { $_.name -eq "Clickup Channel" }).value $Result.onedrive_links[$task.name] = @{ onedrive = $oneDrive clickup_project = $clickupProj clickup_channel = $clickupChan } $shortName = if ($task.name.Length -gt 50) { $task.name.Substring(0,50) + "..." } else { $task.name } if ($oneDrive) { Write-Host " ✓ $shortName" -ForegroundColor Green } else { Write-Host " ⚠ $shortName (no OneDrive)" -ForegroundColor DarkYellow } Start-Sleep -Milliseconds 100 } } catch { Write-Host " ✗ ClickUp fetch failed: $_" -ForegroundColor Red Write-Host " Check your API token + internet connection." -ForegroundColor Red } Write-Host "" # === PART 2: Microsoft Teams Channel Deep-Links === Write-Host "[2/2] Fetching Teams channel deep-links..." -ForegroundColor Yellow try { # Connect to Microsoft Graph (uses existing session if already connected) if (-not (Get-MgContext)) { Write-Host " Connecting to Microsoft Graph (browser will open)..." -ForegroundColor Cyan Connect-MgGraph -Scopes "Team.ReadBasic.All", "Channel.ReadBasic.All", "Group.Read.All" -NoWelcome } # Get all Teams the user is a member of $teamsList = Get-MgUserJoinedTeam -UserId (Get-MgContext).Account -All Write-Host " Found $($teamsList.Count) Teams" -ForegroundColor Green foreach ($team in $teamsList) { try { $channels = Get-MgTeamChannel -TeamId $team.Id -All $Result.teams_channels[$team.DisplayName] = @{} foreach ($channel in $channels) { $Result.teams_channels[$team.DisplayName][$channel.DisplayName] = $channel.WebUrl } Write-Host " ✓ $($team.DisplayName) — $($channels.Count) channels" -ForegroundColor Green } catch { Write-Host " ⚠ Could not read channels in: $($team.DisplayName)" -ForegroundColor DarkYellow } Start-Sleep -Milliseconds 200 } } catch { Write-Host " ✗ Microsoft Graph fetch failed: $_" -ForegroundColor Red Write-Host " Run: Connect-MgGraph -Scopes 'Team.ReadBasic.All','Channel.ReadBasic.All','Group.Read.All'" -ForegroundColor Red Write-Host " Then re-run this script." -ForegroundColor Red } # === SAVE OUTPUT === $Result | ConvertTo-Json -Depth 10 | Out-File -FilePath $OutputFile -Encoding UTF8 Write-Host "" Write-Host "============================================" -ForegroundColor Cyan Write-Host " DONE ✅" -ForegroundColor Green Write-Host "============================================" -ForegroundColor Cyan Write-Host "" Write-Host "Output file: $OutputFile" -ForegroundColor Yellow Write-Host "" Write-Host "Next: open this file, copy ALL content, paste back to Claude." -ForegroundColor Yellow Write-Host "" # Open the file Invoke-Item $OutputFile
